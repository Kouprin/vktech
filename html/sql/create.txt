create database users;
use users;
create table customers (id int auto_increment, global_id int, orders_created int, money_in_orders varchar(50), registered DATETIME, gain varchar(50), primary key (id), index (global_id), index(registered));
create table executors (id int auto_increment, global_id int, orders_completed int, money_received varchar(50), registered DATETIME, gain varchar(50), primary key (id), index (global_id), index(registered));
create database interactions;
use interactions;
create table orders (id int auto_increment, customer_id int, executor_id int, status varchar(20), description varchar(8000), money_cost varchar(50), original_currency char(3), created DATETIME, last_action DATETIME, primary key (id), index (customer_id), index (executor_id), index(last_action));
